id int AUTO_INCREMENT,
title varchar(60) NOT NULL,
body longtext NOT NULL,
autor varchar(60) NOT NULL,
created_at timestamp NOT NULL DEFAULT CURRENT_TIMESTAMP,
PRIMARY KEY (id)
);
